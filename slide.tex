\section{Introduction}

\begin{frame}{Motivation}

\textbf{Challenges in Research Collaboration:}

\begin{itemize}
    \item Fragmented tools: Slack for chat, Mendeley for papers, ChatGPT for AI
    \item Lack of context-aware AI assistance
    \item Information overload from exponential literature growth
    \item Data privacy concerns with third-party AI services
    \item Poor real-time collaboration features
\end{itemize}

\vspace{0.3cm}

\textbf{Solution: OpenResearch}
\begin{itemize}
    \item Unified platform integrating chat, papers, and AI
    \item Group-isolated RAG with PostgreSQL + pgvector
    \item Real-time collaboration with Socket.IO
    \item Modern tech stack: Next.js 16, Node.js 20, FastAPI
\end{itemize}

\end{frame}

\begin{frame}{Project Overview}

\textbf{OpenResearch} - Collaboration-first research platform with AI-powered features

\vspace{0.3cm}

\textbf{Key Features:}
\begin{itemize}
    \item Real-time chat with typing indicators
    \item AI chat with @ai trigger (Llama 3.3 70B)
    \item Paper Q\&A and summarization
    \item Semantic search (pgvector + HNSW)
    \item Group-isolated context (no data leakage)
    \item PDF report generation
    \item arXiv integration for paper discovery
\end{itemize}

\vspace{0.3cm}

\textbf{Tech Stack:}
Next.js 16, React 19, Node.js 20, FastAPI, PostgreSQL 16 + pgvector, Groq, OpenAI Embeddings

\end{frame}


\section{Architecture}

\begin{frame}{System Architecture}

\begin{figure}
    \centering
    \includegraphics[width=10cm, height=6cm]{pic/architecture.png}
    \caption{Microservices Architecture}

\textbf{Four Main Components:}
\begin{itemize}
    \item \textbf{Client:} Next.js 16 + React 19 + Socket.IO Client
    \item \textbf{Server:} Node.js 20 + Express + Socket.IO Server
    \item \textbf{AI Service:} FastAPI + Groq + OpenAI Embeddings
    \item \textbf{Database:} PostgreSQL 16 + pgvector Extension
\end{itemize}

\end{frame}

\begin{frame}{Database Schema}

\textbf{Core Tables (15 total):}

\begin{itemize}
    \item \textbf{User Management:} users, invitations
    \item \textbf{Groups:} groups, group\_members
    \item \textbf{Discussions:} sessions, messages, typing\_indicators
    \item \textbf{Papers:} papers, saved\_papers, group\_papers, session\_papers
    \item \textbf{AI \& Vectors:} group\_paper\_vectors, ai\_artifacts, group\_memory\_notes, group\_reports
\end{itemize}

\vspace{0.3cm}

\textbf{Key Design Decisions:}
\begin{itemize}
    \item Every table has group\_id for isolation
    \item HNSW index on embeddings (1536-dim)
    \item Cascade delete for referential integrity
    \item JSONB for flexible metadata
\end{itemize}

\end{frame}

\begin{frame}{Vector Search with pgvector}

\textbf{Cosine Similarity:}

$$
\text{similarity}(\vec{a}, \vec{b}) = \frac{\vec{a} \cdot \vec{b}}{\|\vec{a}\| \|\vec{b}\|}
$$

\textbf{HNSW Index:}
\begin{itemize}
    \item Hierarchical Navigable Small World graph
    \item $O(\log n)$ search complexity
    \item 95-98\% recall with sub-millisecond queries
    \item Multi-layer graph structure
\end{itemize}

\vspace{0.3cm}

\textbf{Query Example:}
\begin{verbatim}
SELECT content, 1-(embedding<=>$1) as score
FROM group_paper_vectors
WHERE group_id = $2
ORDER BY score DESC LIMIT 5
\end{verbatim}

\end{frame}


\section{AI Features}

\begin{frame}{Retrieval-Augmented Generation (RAG)}

\textbf{RAG Pipeline:}

\begin{enumerate}
    \item \textbf{Retrieval Stage:}
    \begin{itemize}
        \item Embed user query (1536-dim vector)
        \item Search group\_paper\_vectors with HNSW
        \item Retrieve top-k relevant documents
    \end{itemize}
    
    \item \textbf{Generation Stage:}
    \begin{itemize}
        \item Construct prompt with retrieved context
        \item Send to Groq Llama 3.3 70B
        \item Stream response tokens to client
        \item Store Q\&A for future retrieval
    \end{itemize}
\end{enumerate}

\vspace{0.3cm}

\textbf{Benefits:}
\begin{itemize}
    \item Reduces hallucinations (94\% factual accuracy)
    \item Provides source citations
    \item Context-aware responses
\end{itemize}

\end{frame}

\begin{frame}{AI-Powered Features}

\textbf{1. Paper Summarization}
\begin{itemize}
    \item Extracts key points and contributions
    \item Generated in 3-4 seconds
    \item Stored for future context
\end{itemize}

\vspace{0.2cm}

\textbf{2. Paper Q\&A}
\begin{itemize}
    \item Ask questions about specific papers
    \item RAG retrieves paper content + summaries
    \item Provides detailed answers with citations
\end{itemize}

\vspace{0.2cm}

\textbf{3. Group AI Chat}
\begin{itemize}
    \item Conversational AI with full group context
    \item Triggered by @ai mention
    \item Searches across all group papers
\end{itemize}

\vspace{0.2cm}

\textbf{4. Semantic Search}
\begin{itemize}
    \item Conceptual paper discovery
    \item Vector similarity with HNSW
\end{itemize}

\end{frame}

\begin{frame}{Group Isolation Architecture}

\textbf{Key Principle:} Each group = isolated context space

\vspace{0.3cm}

\textbf{Implementation:}
\begin{itemize}
    \item All queries filtered by group\_id
    \item Foreign key constraints enforce isolation
    \item Vector search scoped to group
    \item No cross-group data leakage
\end{itemize}

\vspace{0.3cm}

\textbf{Validation Test:}
\begin{itemize}
    \item Created 2 groups with distinct papers
    \item Asked identical questions in both
    \item \textbf{Result:} 100\% isolation verified
    \item No references to other group's papers
\end{itemize}

\vspace{0.3cm}

\textbf{Security Benefits:}
\begin{itemize}
    \item Data privacy between research teams
    \item Compliant with confidentiality requirements
\end{itemize}

\end{frame}


\section{Implementation}

\begin{frame}{Real-Time Communication}

\textbf{Socket.IO Architecture:}

\begin{itemize}
    \item Bidirectional event-based communication
    \item Room-based message distribution
    \item Automatic reconnection with backoff
    \item WebSocket + HTTP long-polling fallback
\end{itemize}

\vspace{0.3cm}

\textbf{Key Events:}
\begin{itemize}
    \item \texttt{message:send} - Client sends message
    \item \texttt{message:new} - Server broadcasts to room
    \item \texttt{typing:start/stop} - Typing indicators
    \item \texttt{user:join/leave} - Presence tracking
    \item \texttt{message:ai} - AI response streaming
\end{itemize}

\vspace{0.3cm}

\textbf{Performance:}
\begin{itemize}
    \item Median latency: 18ms
    \item 95th percentile: 42ms
\end{itemize}

\end{frame}

\begin{frame}{Authentication \& Security}

\textbf{JWT-Based Authentication:}
\begin{itemize}
    \item Access tokens (15-min expiry)
    \item Refresh tokens (7-day expiry)
    \item Automatic token refresh
    \item Stateless authentication
\end{itemize}

\vspace{0.3cm}

\textbf{Security Measures:}
\begin{itemize}
    \item Bcrypt password hashing (10 rounds)
    \item Role-based access control (Owner/Member)
    \item Input validation and sanitization
    \item SQL injection prevention (parameterized queries)
    \item CORS configuration
    \item Environment variable secrets
\end{itemize}

\vspace{0.3cm}

\textbf{Authorization:}
\begin{itemize}
    \item Middleware validates group membership
    \item Role checks for sensitive operations
\end{itemize}

\end{frame}
